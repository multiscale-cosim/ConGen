
###############################################################
# Create and install python programs using cmake and setup.py
# Based on:
# https://bloerg.net/2012/11/10/cmake-and-distutils.html
# 
# - Creates a versioned python setup.exe
# - packs all files and compiles them
# - make install installs in the CMAKE_INSTALL_PREFIX
#
# TODO:
#  - I'm not sure if dependencys are correctly checked: Stale data
#    might be an issue
#
# Usage: Add the directory you want to include in the build in:
#        setup.py.in  (marked clearly)


find_program(PYTHON "python")
if (PYTHON)
    set(SETUP_PY_IN "${CMAKE_CURRENT_SOURCE_DIR}/setup.py.in")
    set(SETUP_PY    "${CMAKE_CURRENT_BINARY_DIR}/setup.py")
    set(DEPS        "${CMAKE_CURRENT_SOURCE_DIR}/vicogen/__init__.py")
    set(OUTPUT      "${CMAKE_CURRENT_BINARY_DIR}/build/timestamp")

    configure_file(${SETUP_PY_IN} ${SETUP_PY})

    add_custom_command(OUTPUT ${OUTPUT}
                       COMMAND ${PYTHON} ${SETUP_PY} build
                       COMMAND ${CMAKE_COMMAND} -E touch ${OUTPUT}
                       DEPENDS ${DEPS})

    add_custom_target(target ALL DEPENDS ${OUTPUT})

    install(CODE "execute_process(COMMAND ${PYTHON} ${SETUP_PY} install -f --prefix=${CMAKE_INSTALL_PREFIX})")
endif()
